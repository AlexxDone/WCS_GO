block spawn
{
	es wcsgroup set jack event_var(userid) 1
	es wcs_getweapon event_var(userid) wcs_tmp 1
	if (server_var(wcs_tmp) == -1) do
	{
		es_give event_var(userid) weapon_ak47
	}
	es wcs_getweapon event_var(userid) wcs_tmp 2
	if (server_var(wcs_tmp) != weapon_deagle) do
	{
		es removeweapon event_var(userid) 2
		es_give event_var(userid) weapon_deagle
	}
	es_playsound event_var(userid) weapons\rpg\rocketfire1.wav 1
	es_getplayerlocation wcs_x1 wcs_y1 wcs_z1 event_var(userid)
	es est_effect 10 #a 0 sprites/smoke.vmt server_var(wcs_x1) server_var(wcs_y1) server_var(wcs_z1) 99 50 2 3 11 0 25 25 25 255 9
	es_xmath wcs_z1 + 10
	es_delayed 0.2 est_effect 10 #a 0 sprites/smoke.vmt server_var(wcs_x1) server_var(wcs_y1) server_var(wcs_z1) 99 50 2 3 11 0 0 255 0 255 9
	es_xmath wcs_z1 + 10
	es_delayed 0.4 est_effect 10 #a 0 sprites/smoke.vmt server_var(wcs_x1) server_var(wcs_y1) server_var(wcs_z1) 99 50 2 3 11 0 25 25 25 255 9
	es_xmath wcs_z1 + 10
	es_delayed 0.6 est_effect 10 #a 0 sprites/smoke.vmt server_var(wcs_x1) server_var(wcs_y1) server_var(wcs_z1) 99 50 2 3 11 0 0 255 0 255 9
}

block death
{
	es wcsgroup set jack event_var(userid) 0
}

block load
{
	es_xset wcs_z1 0
	es_xset wcs_y1 0
	es_xset wcs_x1 0
	es_xset wcs_z2 0
	es_xset wcs_y2 0
	es_xset wcs_x2 0
	es_xset wcs_dmg 0
}
block onchange
{
	es wcsgroup set jack server_var(wcs_userid) 0
}

block damage
{
	es_rand wcs_dmg 5 18
	es wcs_dealdamage event_var(userid) event_var(attacker) server_var(wcs_dmg)
	es_tell event_var(attacker) #multi #green[WCS] #lightgreenYou hit a vital spot and did #green server_var(wcs_dmg) #lightgreendamage to #green event_var(es_username)
}

block longjump
{
	es wcs_setfx longjump event_var(userid) = server_var(wcs_multiplier) 0
	es_tell event_var(userid) #multi #green[WCS]#lightgreen Hard exercise - Increased your jump length
}

block push
{
	es wcs_pushed event_var(userid) 0 0 350
	es_tell event_var(userid) #multi #green[WCS]#lightgreen Shook - You were launched into the air
	es_tell event_var(attacker) #multi #green[WCS]#lightgreen Shook - You launched your enemy into the air
	es_getplayerlocation wcs_x1 wcs_y1 wcs_z1 event_var(userid)

block swap
{
	es wcsgroup get jack wcs_tmp2 server_var(wcs_userid)
	if (server_var(wcs_tmp2) = 1) do
	{
		es removeweapon server_var(wcs_userid) 1
		es removeweapon server_var(wcs_userid) 2
		es_give server_var(wcs_userid) weapon_ssg08
		es_give server_var(wcs_userid) weapon_deagle
		es wcsgroup set jack server_var(wcs_userid) 2
		es_tell server_var(wcs_userid) #multi #green [WCS]#lightgreen Weapon changer - Switched to scout
	}
	if (server_var(wcs_tmp2) = 2) do
	{
		es removeweapon server_var(wcs_userid) 1
		es removeweapon server_var(wcs_userid) 2
		es_give server_var(wcs_userid) weapon_ak47
		es_give server_var(wcs_userid) weapon_deagle
		es wcsgroup set jack server_var(wcs_userid) 1
		es_tell server_var(wcs_userid) #multi #green [WCS]#lightgreen Weapon changer - Switched to an AK47
	}
}
