block empower
{
	es wcs_get_skill_level event_var(userid) wcs_tmp1 1
	es_xset wcs_addhealth 5
	es_math wcs_addhealth * server_var(wcs_tmp1)
	es_xset wcs_speed .05
	es_math wcs_speed * server_var(wcs_tmp1)
	es_math wcs_speed + 1
	es_xset wcs_multiplier .8
	es_math wcs_multiplier * server_var(wcs_tmp1)
	es_xset wcs_tmp3 1
	es_xset wcs_gravity .07
	es_math wcs_gravity * server_var(wcs_tmp1)
	es_math wcs_tmp3 - server_var(wcs_gravity)
	es_xrand wcs_rand 1 6
	if (server_var(wcs_rand) == 1) do
	{
		es wcs_setfx health event_var(userid) + server_var(wcs_addhealth) 0
		es_tell event_var(userid) #multi #green[WCS]#lightgreen Empower - Increased your health by server_var(wcs_addhealth)
	}
	if (server_var(wcs_rand) == 2) do
	{
		es wcs_setfx speed event_var(userid) = server_var(wcs_speed) 0
		es_set wcs_speed_var server_var(wcs_speed)
		es_xmath wcs_speed_var * 100
		es_xmath wcs_speed_var - 100
		es wcs_decimal wcs_speed_var server_var(wcs_speed_var)
		es_tell event_var(userid) #multi #green[WCS]#lightgreen Empower - Increased your speed by server_var(wcs_speed_var)%
	}
	if (server_var(wcs_rand) == 3) do
	{
		es wcs_setfx longjump event_var(userid) = server_var(wcs_multiplier) 0
		es_xmath wcs_multiplier * 100
		es_tell event_var(userid) #multi #green [WCS]: #lightgreen Empower - Increased your jump lenght by server_var(wcs_multiplier)
	}
	if (server_var(wcs_rand) == 4) do
	{
		es wcs_setfx gravity event_var(userid) = server_var(wcs_tmp3) 0
		es_xmath wcs_tmp3 * 100
		es_tell event_var(userid) #multi #green[WCS]#lightgreen Empower - Reduced your gravity to server_var(wcs_tmp3)%
	}
	if (server_var(wcs_rand) == 5) do
	{
		es wcs_domole event_var(userid) 5
		es_centertell event_var(userid) Empower teleports you to the enemy spawn in 5 seconds!
	}
	if (server_var(wcs_rand) == 6) do
	{
		es wcs_get_skill_level event_var(userid) wcs_tmp1 1
		es_xset wcs_addhealth 5
		es_math wcs_addhealth * server_var(wcs_tmp1)
		es_math wcs_addhealth + 100
		es_tell event_var(userid) #multi #green[WCS]#lightgreen Empower - Recovers health over time
		es_xrand wcs_rand 3 15
		es_xrand wcs_duration 3 8
		es_delayed 1 wcs_regeneration event_var(userid) server_var(wcs_rand) server_var(wcs_duration) server_var(wcs_addhealth) 1000 250
	}
}
